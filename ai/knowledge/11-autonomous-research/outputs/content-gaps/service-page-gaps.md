```astro

  <!-- Hero Section -->
  <ServiceHero
    title="Service Name"
    description="Brief overview"
    price="Starting at $X"
    phone="(907) 310-3000"
    cta="Get Free Estimate"
  />

  <!-- Trust Bar -->
  <TrustBar
    items={["Licensed & Insured", "10+ Years", "1000+ Customers"]}
  />

  <!-- Main Content -->
  <section>
    <h2>Why [Service] for Alaska Homes</h2>
    <!-- 300-500 words of Alaska-specific content -->
  </section>

  <!-- Process Steps -->
  <ProcessSteps steps={[...]} />

  <!-- Benefits Grid -->
  <BenefitsGrid benefits={[...]} />

  <!-- Before/After Gallery -->
  <BeforeAfterGallery images={[...]} />

  <!-- Testimonials -->
  <TestimonialSlider testimonials={[...]} />

  <!-- FAQ Section -->
  <FAQSection faqs={[...]} /> <!-- With schema -->

  <!-- Related Services -->
  <RelatedServices services={[...]} />

  <!-- CTA Section -->
  <CTASection
    title="Ready to Get Started?"
    phone="(907) 310-3000"
  />

  <!-- Contact Form -->

</ServicePageLayout>
```
